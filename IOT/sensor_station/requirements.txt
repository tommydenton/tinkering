# Sensor Station Requirements
# Install with: pip install -r requirements.txt

# Core Blinka library for CircuitPython on Linux
adafruit-blinka

# Display library - use RGB Display (PIL-based), NOT adafruit-circuitpython-st7789
# The standalone st7789 library is displayio-based and conflicts with PIL usage
adafruit-circuitpython-rgb-display

# Sensor libraries
adafruit-circuitpython-ds3231      # DS3231 RTC
adafruit-circuitpython-ahtx0       # AHT20 Temperature/Humidity
adafruit-circuitpython-bmp280      # BMP280 Pressure/Altitude
adafruit-circuitpython-scd4x       # SCD-41 CO2 Sensor
adafruit-circuitpython-gps         # PA1010D GPS

# PIL for display graphics (usually available via system packages)
# pillow
